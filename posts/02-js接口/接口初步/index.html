百度智能提示的默认回调函数 window.baidu.sug

接口的使用

原始示例

  • 各大搜索引擎智能提示API（JSONP跨域实现自动补全搜索建议） - woider - 博客园 (cnblogs.com)

API返回的是JSONP数据，JSONP是跨域访问的一种方式。由于服务器返回的JavaScript代码可以直接引用，通过回调函数的方式就可以间接的获取服务器的数据。

window.baidu.sug 返回的是一个json对象

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <script>

        window.onload = function () {

            //组装查询地址
            var sugurl = "http://suggestion.baidu.com/su?wd=#content#&cb=window.baidu.sug";
            var content = "关键词";
            sugurl = sugurl.replace("#content#", content);

            //定义回调函数
            window.baidu = {
                sug: function (json) {
                    console.log(json.s)
                    //编写代码进行dom拼装
                }
            }

            //动态添加JS脚本
            var script = document.createElement("script");
            script.src = sugurl;
            document.getElementsByTagName("head")[0].appendChild(script);

        }

    </script>
</head>

<body>

</body>

</html>

控制台打印的结果：如果要将结果保存在一个字符串数组中，只需要 var arr = json.s 即可。

是否需要用jquery?

  • 对于已经编写好的类似百度接口自带回调函数的情况，不用任何ajax，也不需要jquery。直接就是一个html就可以实现接口掉用。

完整代码

<html lang="en">

<head>
    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <script>

        window.onload = function () {
            var ot = document.getElementById('t1');
            ot.oninput = function () {
                console.log(ot.value);
                //组装查询地址
                // var sugurl = "http://suggestion.baidu.com/su?wd=#content#&cb=show";
                var sugurl = "http://sp0.baidu.com/5a1Fazu8AA54nxGko9WTAnF6hhy/su?wd=#content#&cb=show";
                var content = ot.value;
                sugurl = sugurl.replace("#content#", content);
                console.log("sugurl:", sugurl);
                //定义回调函数
                show = function (json) {
                    console.log(json.s)
                }
                //动态添加JS脚本
                var script = document.createElement("script");
                script.src = sugurl;
                document.getElementsByTagName("head")[0].appendChild(script);
            }
        }

</head>

</html>

    window.onload = function () {
        minAjax({
            // url: "https://v.api.aa1.cn/api/yiyan/index.php",//request URL
            //更改为我们自己的同源接口

                // alert(data);
                // console.log(data);
                //拼装dom
                var divd=document.getElementById('d1');
                divd.innerHTML=data;
            }

        });
    }
</script>

<body>

</body>

</html>
